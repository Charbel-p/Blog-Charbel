AVIS

SELECT a.id, a.note, a.texte_avis, u.nom, ar.titre
FROM avis a
INNER JOIN utilisateurs u ON a.utilisateur_id = u.id
INNER JOIN articles ar ON a.article_id = ar.id;

SELECT article_id, ROUND(AVG(note), 1) AS note_moyenne, COUNT(*) AS nombre_avis
FROM avis
WHERE article_id = 1
GROUP BY article_id;

SELECT c.id, c.contenu, COUNT(cl.id) AS nb_likes
FROM commentaires c
LEFT JOIN commentaire_likes cl ON c.id = cl.commentaire_id
GROUP BY c.id, c.contenu
ORDER BY nb_likes DESC;


STATISTIQUES

SELECT 'utilisateurs' AS table_nom, COUNT(*) AS total FROM utilisateurs
UNION ALL
SELECT 'categories', COUNT(*) FROM categories
UNION ALL
SELECT 'articles', COUNT(*) FROM articles
UNION ALL
SELECT 'commentaires', COUNT(*) FROM commentaires
UNION ALL
SELECT 'avis', COUNT(*) FROM avis
UNION ALL
SELECT 'commentaire_likes', COUNT(*) FROM commentaire_likes;

SELECT
    a.titre,
    c.nom AS categorie,
    COUNT(DISTINCT co.id) AS nb_commentaires,
    ROUND(AVG(av.note), 1) AS note_moyenne
FROM articles a
INNER JOIN categories c ON a.categorie_id = c.id
LEFT JOIN commentaires co ON a.id = co.article_id AND co.est_approuve = 1
LEFT JOIN avis av ON a.id = av.article_id
WHERE a.est_publie = 1
GROUP BY a.id, a.titre, c.nom
ORDER BY a.publie_le DESC;

SELECT u.nom, u.courriel,
       (SELECT COUNT(*) FROM commentaires WHERE utilisateur_id = u.id) AS commentaires,
       (SELECT COUNT(*) FROM avis WHERE utilisateur_id = u.id) AS notes,
       (SELECT COUNT(*) FROM commentaire_likes WHERE utilisateur_id = u.id) AS likes_donnes
FROM utilisateurs u
WHERE u.id = 2;
